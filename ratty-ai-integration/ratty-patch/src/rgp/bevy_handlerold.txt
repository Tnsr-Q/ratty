// ─── NEW: PROCESS VISUALIZATION ───

#[derive(Component)]
pub struct ProcessOrb {
    pub pid: u32,
    pub cpu_usage: f32,
    pub mem_usage: f32,
}

fn handle_ps(
    mut events: EventReader<RattyAiCommand>,
    mut commands: Commands,
    mut meshes: ResMut<Assets<Mesh>>,
    mut materials: ResMut<Assets<StandardMaterial>>,
) {
    for event in events.read() {
        if let RattyAiCommand::Ps { visualize, highlight, color } = event {
            if *visualize {
                info!("Ratty AI: Visualizing processes");
                // Spawn process orbs — size = memory, brightness = CPU
                // Highlighted PID glows with specified color
            }
        }
    }
}

fn handle_kill(
    mut events: EventReader<RattyAiCommand>,
    mut commands: Commands,
    query: Query<(Entity, &ProcessOrb)>,
) {
    for event in events.read() {
        if let RattyAiCommand::Kill { pid, effect } = event {
            for (entity, orb) in query.iter() {
                if orb.pid == *pid {
                    match effect.as_str() {
                        "explode" => {
                            // Spawn particle explosion at orb position
                            info!("Ratty AI: Process {} explodes!", pid);
                        }
                        "shrink" => {
                            // Animate scale to zero
                            info!("Ratty AI: Process {} shrinks away", pid);
                        }
                        "dissolve" => {
                            // Fade out with noise
                            info!("Ratty AI: Process {} dissolves", pid);
                        }
                        _ => {}
                    }
                    commands.entity(entity).despawn_recursive();
                }
            }
        }
    }
}

// ─── NEW: AI STATE INDICATORS ───

#[derive(Resource, Default)]
pub struct AiState {
    pub thinking: bool,
    pub confidence: f32,
    pub mood: String,
}

fn handle_ai_state(
    mut events: EventReader<RattyAiCommand>,
    mut state: ResMut<AiState>,
) {
    for event in events.read() {
        match event {
            RattyAiCommand::Think { state: s } => {
                state.thinking = match s.as_str() {
                    "start" => true,
                    "end" => false,
                    _ => !state.thinking,
                };
                info!("Ratty AI: Thinking = {}", state.thinking);
                // Cursor pulses slowly, faint ambient sound
            }
            RattyAiCommand::Confidence { level } => {
                state.confidence = *level;
                info!("Ratty AI: Confidence = {:.0}%", level * 100.0);
                // Green aura (high) → Yellow (medium) → Red (low)
            }
            RattyAiCommand::Mood { mood } => {
                state.mood = mood.clone();
                info!("Ratty AI: Mood = {}", mood);
                match mood.as_str() {
                    "excited" => { /* Fast spin, bright colors */ }
                    "cautious" => { /* Slow, dim, careful */ }
                    "confused" => { /* Erratic warp, tilted */ }
                    "focused" => { /* Steady, minimal motion */ }
                    "celebratory" => { /* Confetti, fast spin */ }
                    _ => {}
                }
            }
            _ => {}
        }
    }
}

    mut events: EventReader<RattyAiCommand>,
    mut recorder: ResMut<MacroRecorder>,
) {
    for event in events.read() {
        match event {
            RattyAiCommand::MacroRecord { name } => {
                recorder.recording = true;

                recorder.buffer.clear();
                info!("Ratty AI: Recording macro '{}'", name);

                info!("Ratty AI: Playing macro '{}'", name);
                // Replay recorded OSC sequences
            }
            RattyAiCommand::MacroExport { name, to } => {
                info!("Ratty AI: Export macro '{}' to {}", name, to);
                // Write buffer to .ratty file
            }
            RattyAiCommand::MacroRun { path } => {
                info!("Ratty AI: Running macro {}", path);
                // Read .ratty file and execute each line
            }
            _ => {}
        }
    }
}

// ─── NEW: REACTIVE ───

#[derive(Resource, Default)]
pub struct ReactiveTriggers {
    pub cpu_threshold: Option<f32>,
    pub memory_threshold: Option<f32>,
    pub battery_threshold: Option<f32>,
    pub effect: String,
}

fn handle_reactive(
    mut events: EventReader<RattyAiCommand>,
    mut triggers: ResMut<ReactiveTriggers>,
) {
    for event in events.read() {
        if let RattyAiCommand::React { effect, cpu_high, memory_high, battery_low } = event {
            triggers.effect = effect.clone();
            triggers.cpu_threshold = *cpu_high;
            triggers.memory_threshold = *memory_high;
            triggers.battery_threshold = *battery_low;
            info!("Ratty AI: Reactive effect '{}' registered", effect);
        }
    }
}
